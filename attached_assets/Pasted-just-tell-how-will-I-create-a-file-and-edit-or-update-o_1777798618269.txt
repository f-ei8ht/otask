Create a File
rust
use std::fs;
use std::path::Path;

fn create_file(path: &str, content: &str, cwd: &str) -> String {
    let full_path = Path::new(cwd).join(path);
    
    // Create parent dirs if needed
    if let Some(parent) = full_path.parent() {
        fs::create_dir_all(parent).ok();
    }
    
    match fs::write(&full_path, content) {
        Ok(_) => format!("Created: {}", full_path.display()),
        Err(e) => format!("Error: {}", e),
    }
}
Edit / Update a File (str_replace style)
rust
fn edit_file(path: &str, old_str: &str, new_str: &str, cwd: &str) -> String {
    let full_path = Path::new(cwd).join(path);

    let content = match fs::read_to_string(&full_path) {
        Ok(c) => c,
        Err(e) => return format!("Error reading file: {}", e),
    };

    let count = content.matches(old_str).count();
    if count == 0 {
        return "Error: old_str not found in file".to_string();
    }
    if count > 1 {
        return "Error: old_str found multiple times, make it more unique".to_string();
    }

    let updated = content.replacen(old_str, new_str, 1);

    match fs::write(&full_path, updated) {
        Ok(_) => format!("Edited: {}", full_path.display()),
        Err(e) => format!("Error writing file: {}", e),
    }
}
Hook into your agent's tool handler
rust
fn execute_tool(name: &str, args: &serde_json::Value, cwd: &str) -> String {
    match name {
        "create_file" => create_file(
            args["path"].as_str().unwrap(),
            args["content"].as_str().unwrap(),
            cwd,
        ),
        "edit_file" => edit_file(
            args["path"].as_str().unwrap(),
            args["old_str"].as_str().unwrap(),
            args["new_str"].as_str().unwrap(),
            cwd,
        ),
        _ => "Unknown tool".to_string(),
    }
}
