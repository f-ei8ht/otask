1. Setup highlighter (do once, not per frame)
rust
use syntect::easy::HighlightLines;
use syntect::highlighting::{ThemeSet, Style as SyntectStyle};
use syntect::parsing::SyntaxSet;
use syntect::util::LinesWithEndings;

pub struct Highlighter {
    pub ps: SyntaxSet,
    pub ts: ThemeSet,
}

impl Highlighter {
    pub fn new() -> Self {
        Self {
            ps: SyntaxSet::load_defaults_newlines(),
            ts: ThemeSet::load_defaults(),
        }
    }
}
2. Convert syntect colors → Ratatui Line
rust
use ratatui::style::{Color, Style};
use ratatui::text::{Line, Span};

pub fn highlight_code<'a>(
    hl: &Highlighter,
    code: &str,
    extension: &str,       // "rs", "py", "js" etc
    scroll_offset: usize,
    visible_lines: usize,
) -> Vec<Line<'a>> {
    let syntax = hl.ps
        .find_syntax_by_extension(extension)
        .unwrap_or_else(|| hl.ps.find_syntax_plain_text());

    // "base16-ocean.dark" | "base16-mocha.dark" | "Solarized (dark)"
    let theme = &hl.ts.themes["base16-ocean.dark"];
    let mut h = HighlightLines::new(syntax, theme);

    let lines: Vec<&str> = code.lines().collect();

    lines
        .iter()
        .enumerate()
        .skip(scroll_offset)
        .take(visible_lines)
        .map(|(i, line)| {
            let ranges = h.highlight_line(line, &hl.ps).unwrap_or_default();

            let mut spans = vec![
                // line number gutter
                Span::styled(
                    format!("{:>4} │ ", i + 1),
                    Style::new().fg(Color::Rgb(108, 112, 134)), // overlay gray
                ),
            ];

            for (style, text) in ranges {
                let fg = syntect_to_ratatui(style);
                spans.push(Span::styled(text.to_string(), Style::new().fg(fg)));
            }

            Line::from(spans)
        })
        .collect()
}

fn syntect_to_ratatui(style: SyntectStyle) -> Color {
    let c = style.foreground;
    Color::Rgb(c.r, c.g, c.b)
}
3. Render in your editor panel
rust
fn render_editor(f: &mut Frame, app: &mut App, area: Rect, hl: &Highlighter) {
    let ext = app.selected_file
        .as_deref()
        .and_then(|p| p.rsplit('.').next())
        .unwrap_or("txt");

    let visible = area.height.saturating_sub(2) as usize; // minus borders

    let lines = highlight_code(
        hl,
        &app.file_content,
        ext,
        app.scroll,
        visible,
    );

    let filename = app.selected_file.as_deref().unwrap_or("untitled");

    f.render_widget(
        Paragraph::new(lines)
            .block(
                Block::bordered()
                    .title(format!("  {filename}"))
                    .border_type(BorderType::Rounded)
                    .border_style(Style::new().fg(Color::Rgb(49, 50, 68)))
                    .bg(Color::Rgb(30, 30, 46)),
            ),
        area,
    );
}
4. Scroll state
rust
// in App
pub scroll: usize,

// key bindings
KeyCode::Char('j') | KeyCode::Down => {
    let max = app.file_content.lines().count().saturating_sub(1);
    if app.scroll < max { app.scroll += 1; }
}
KeyCode::Char('k') | KeyCode::Up => {
    if app.scroll > 0 { app.scroll -= 1; }
}
KeyCode::Char('d') if ctrl => { app.scroll += 20; } // half page
KeyCode::Char('u') if ctrl => { app.scroll = app.scroll.saturating_sub(20); }
KeyCode::Char('g') => { app.scroll = 0; }            // top
KeyCode::Char('G') => {                               // bottom
    app.scroll = app.file_content.lines().count().saturating_sub(1);
}
5. Load file when selected from tree
rust
KeyCode::Enter => {
    if let Some(path) = app.tree_state.selected().last() {
        if std::path::Path::new(path).is_file() {
            if let Ok(content) = std::fs::read_to_string(path) {
                app.file_content = content;
                app.selected_file = Some(path.clone());
                app.scroll = 0; // reset scroll
            }
        }
    }
}

Performance tip
HighlightLines reprocesses from line 1 every render — for large files cache the highlighted lines and only re-highlight on file change:

rust
pub struct App {
    pub highlighted_cache: Vec<Line<'static>>,
    pub cache_dirty: bool,
}
